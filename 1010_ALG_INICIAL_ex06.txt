Var
// Seção de Declarações das variáveis
        base,altura,area : real

Inicio
// Seção de Comandos, procedimento, funções, operadores, etc...
   escreval(":: Calculador de Áreas do Retangulo :: ")
   escreva("Informe um valor para a base: ")
   leia(base)
   escreval("Informe o valor da altura: ")
   leia(altura)
   escreval("Aguarde ...")
   timer 2000
   area <- base * altura
   escreval("Calculando ")
   escreval("A área do retangulo é ",area,"m2.")

Fimalgoritmo
