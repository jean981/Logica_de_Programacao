Var
// Seção de Declarações das variáveis
        n,soma : real
        p, r   : inteiro

Inicio
// Seção de Comandos, procedimento, funções, operadores, etc...
   escreval(":: Calcular o quadrado e o cubo de N :: ")
   escreva("Informe um numero:  ")
   leia(n)
   r <- quad(n)
   escreval("O quadrado de ",n," é ",r)
   p <- exp(n,3)
   escreval("O cubo de ",n," é ",p)
   soma <- p + r
   escreval("A soma dos valores ",r," + ",p," é ",soma)

Fimalgoritmo
