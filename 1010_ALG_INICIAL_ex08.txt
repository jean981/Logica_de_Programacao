Inicio

   leia(_F)

   timer 2000
   _C <- (5 *((_F - 32)/9))
   escreval("Calculando ")
   timer 2000
   escreval("Temperatura convertida: ")
   escreval(_F,"°Fahrenheit.")
   escreval(" ",_C:2,"°Celsius.")
